### Installation

```sh
yarn add @instructure/ui-alerts
```

### Usage

```js
import React from 'react'
import { Alert } from '@instructure/ui-alerts'

const MyAlert = function () {
  return (
    <Alert>Hello alert</Alert>
  )
}
```
